昨天和今天主要学的是线段树，总的来说目前是支持范围增删和范围修改，维护累加和或者最大值
最有意思的是懒更新，主要是递归到被l，r包住的任务时懒住，不在往下传了，记录懒信息。
总的来说，以模板为例，大致流程如下
build：建树
up函数：维护左右子树的最大/累加值
down函数：下发懒信息
add函数
update函数
query函数
还有一些零碎知识比如建树的大小是4n，左右子树i分别是i<<1和i<<1|1
最后附上今日最难题目 hdu2614插花
#include<bits/stdc++.h>
using namespace std;
const int N=50005;
int n;
int sum[N<<2];
int change[N<<2];
bool update[N<<2];
void build(int l,int r,int i){
	if(l<r){
		int mid=(l+r)>>1;
		build(l,mid,i<<1);
		build(mid+1,r,i<<1|1);
	}
	sum[i]=0;
	change[i]=0;
	update[i]=false; 
}
void up(int i){
	sum[i]=sum[i<<1]+sum[i<<1|1];
}
void lazy(int i,int v,int cnt){
	sum[i]=v*cnt;
	change[i]=v;
	update[i]=true;
}
void down(int i,int ln,int rn){
	if(update[i]){
		lazy(i<<1,change[i],ln);
		lazy(i<<1|1,change[i],rn);
		update[i]=false;
	}
}
void Update(int jobl,int jobr,int jobv,int l,int r,int i){
	if(jobl<=l&&jobr>=r){
		lazy(i,jobv,r-l+1);
		return; 
	}
	int mid=(l+r)>>1;
	down(i,mid-l+1,r-mid);
	if(jobl<=mid){
		Update(jobl,jobr,jobv,l,mid,i<<1);
	}
	if(jobr>mid){
		Update(jobl,jobr,jobv,mid+1,r,i<<1|1);
	}
	up(i);
}
int query(int jobl,int jobr,int l,int r,int i){
	if(jobl<=l&&jobr>=r){
		return sum[i];
	}
	int ans=0;
	int mid=(l+r)>>1;
	down(i,mid-l+1,r-mid);
	if(jobl<=mid){
		ans+=query(jobl,jobr,l,mid,i<<1);
	}
	if(jobr>mid){
		ans+=query(jobl,jobr,mid+1,r,i<<1|1);
	}
	return ans;
}
int findzero(int from,int k){
	int ans=0;
	int l=from,r=n,mid;
	while(l<=r){
		int mid=(l+r)>>1;
		if(mid-from+1-query(from,mid,1,n,1)>=k){
			ans=mid;
			r=mid-1;
		}else{
			l=mid+1;
		}
	} 
	return ans;
}
vector<int> insert(int from,int flower){
	from++;
	int start,end;
	int zero=n-from+1-query(from,n,1,n,1);
	if(zero==0){
		start=0;
		end=0;
	}else{
		start=findzero(from,1);
		end=findzero(from,min(zero,flower));
		Update(start,end,1,1,n,1);
	}
	start--;
	end--;
	return {start,end};
}
int clear(int l,int r){
	l++;
	r++;
	int ans=query(l,r,1,n,1);
	Update(l,r,0,1,n,1);
	return ans;
}
