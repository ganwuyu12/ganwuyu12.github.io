一日不见，如隔三秋~
今天主要解决的是解决区间合并问题，多数可以抽象成查询连续1的长度；
这种题目模板就是把一段区间【i】,用len，pre，suf维护，最后能得出连续区间长度，
一共四道，序列操作最全，分别有修改，累加，和连续区段，代码也最长（打起来有点无聊）
最长相连房子，和最长交替子序列差不多，唯一有点不同的就是最左侧连续大于x个空房间，在query时先左再中再右（满足区间一定有大于x的情况时）
下面还是放上序列操作，毕竟最长，维护的也最多
#include<bits/stdc++.h>
using namespace std;
const int N=1e5+5; 
int arr[N];
int sum[N<<2];
int pre0[N<<2],suf0[N<<2],len0[N<<2];
int pre1[N<<2],suf1[N<<2],len1[N<<2];
int change[N<<2];
bool updt[N<<2],re[N<<2];
void up(int i,int ln,int rn){
	int l=i<<1;
	int r=i<<1|1;
	sum[i]=sum[l]+sum[r];
	len1[i]=max({len1[l],len1[r],suf1[l]+pre1[r]});
	len0[i]=max({len0[l],len0[r],suf0[l]+pre0[r]}); 
	pre1[i]=len1[l]<ln?pre1[l]:pre1[l]+pre1[r];
	suf1[i]=len1[r]<rn?suf1[r]:suf1[r]+suf1[l];
	pre0[i]=len0[l]<ln?pre0[l]:pre0[l]+pre0[r];
	suf0[i]=len0[r]<rn?suf0[r]:suf0[r]+suf0[l];
}
void updatelazy(int i,int v,int cnt){
	sum[i]=v*cnt;
	pre1[i]=suf1[i]=len1[i]=v==1?cnt:0;
	pre0[i]=suf0[i]=len0[i]=v==0?cnt:0;
	change[i]=v;
	updt[i]=true;
	re[i]=false;
}
void reverselazy(int i,int cnt){
	sum[i]=cnt-sum[i];
	int tmp=len1[i];len1[i]=len0[i];len0[i]=tmp;
	tmp=pre1[i];pre1[i]=pre0[i];pre0[i]=tmp;
	tmp=suf1[i];suf1[i]=suf0[i];suf0[i]=tmp;
	re[i]=!re[i];
}
void down(int i,int ln,int rn){
	if(updt[i]){
		updatelazy(i<<1,change[i],ln);
		updatelazy(i<<1|1,change[i],rn);
		updt[i]=false;
	}
	if(re[i]){
		reverselazy(i<<1,ln);
		reverselazy(i<<1|1,rn);
		re[i]=!re[i];
	}
}
void update(int jobl,int jobr,int jobv,int l,int r,int i){
	if(jobl<=l&&jobr>=r){
		updatelazy(i,jobv,r-l+1);
	}else{
		int mid=(l+r)>>1;
		down(i,mid-l+1,r-mid);
		if(jobl<=mid){
			update(jobl,jobr,jobv,l,mid,i<<1);
		}
		if(jobr>mid){
			update(jobl,jobr,jobv,mid+1,r,i<<1|1);
		}
		up(i,mid-l+1,r-mid);
	}
}
void rever(int jobl,int jobr,int l,int r,int i){
	if(jobl<=l&&jobr>=r){
		reverselazy(i,r-l+1);
	}else{
		int mid=(l+r)>>1;
		down(i,mid-l+1,r-mid);
		if(jobl<=mid){
			rever(jobl,jobr,l,mid,i<<1);
		}
		if(jobr>mid){
			rever(jobl,jobr,mid+1,r,i<<1|1);
		}
		up(i,mid-l+1,r-mid);
	}
}
int qsum(int jobl,int jobr,int l,int r,int i){
	if(jobl<=l&&jobr>=r) {
		return sum[i];
	} else {
		int ans=0;
		int mid=(l+r)>>1;
		down(i,mid-l+1,r-mid);
		if(jobl<=mid) {
			ans+=qsum(jobl,jobr,l,mid,i<<1);
		}
		if(jobr>mid) {
			ans+=qsum(jobl,jobr,mid+1,r,i<<1|1);
		}
		return ans;
	}
}
vector<int> qlen(int jobl,int jobr,int l,int r,int i){
	if(jobl<=l&&jobr>=r){
		return {len1[i],pre1[i],suf1[i]};
	}else{
		int mid=(l+r)>>1;
		int ln=mid-l+1;
		int rn=r-mid;
		down(i,mid-l+1,r-mid);
		vector<int> ans1,ans2;
		if(jobl<=mid){
			ans1=qlen(jobl,jobr,l,mid,i<<1);
		}
		if(jobr>mid){
			ans2=qlen(jobl,jobr,mid+1,r,i<<1|1);
		}
		if(!ans1.empty()&&!ans2.empty()){
			int anslen=max({ans1[0],ans2[0],ans1[2]+ans2[1]});
			int anspre=ans1[1]<ln?ans1[1]:ans1[1]+ans2[1];
			int anssuf=ans2[2]<rn?ans2[2]:ans2[2]+ans1[2];
			return {anslen,anspre,anssuf};
		}else{
			if(!ans1.empty()){
				return {ans1[0],ans1[1],ans1[2]};
			}else{
				return {ans2[0],ans2[1],ans2[2]};
			}
		}
	}
}
void build(int l,int r,int i){
	if(l==r){
		sum[i]=arr[l];
		len1[i]=pre1[i]=suf1[i]=arr[l]==1?1:0;
		len0[i]=pre0[i]=suf0[i]=arr[l]==0?1:0;
	}else{
		int mid=(l+r)>>1;
		build(l,mid,i<<1);
		build(mid+1,r,i<<1|1);
		up(i,mid-l+1,r-mid);
	}
	updt[i]=false;
	re[i]=false;
}
int main(){
	int n,m;
	cin>>n>>m;
	for(int i=1;i<=n;i++) cin>>arr[i];
	build(1,n,1);
	for(int i=0;i<m;i++){
		int op;cin>>op;
		if(op==0){
			int l,r;
			cin>>l>>r;
			l++,r++;
			update(l,r,0,1,n,1);
		}else if(op==1){
			int l,r;cin>>l>>r;
			l++,r++;
			update(l,r,1,1,n,1);
		}else if(op==2){
			int l,r;cin>>l>>r;
			l++,r++;
			rever(l,r,1,n,1);
		}else if(op==3){
			int l,r;cin>>l>>r;
			l++,r++;
			int ans=qsum(l,r,1,n,1);
			cout<<ans<<endl;
		}else{
			int l,r;cin>>l>>r;
			l++,r++;
			vector<int> ans=qlen(l,r,1,n,1);
			cout<<ans[0]<<endl;
		}
	}
}
