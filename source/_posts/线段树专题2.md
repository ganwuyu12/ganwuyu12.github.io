依旧线段树。
下午一共就写了三题qwq
前两道一个是开方操作，一个是取模操作，显然不符合线段树前提在O（1）条件内的操作，需要暴力修改每一个子树的值
但运用了势能分析，发现两者都是logn内会收敛到1，所以总复杂度没爆
重点是两个都有剪枝，一个是Max【i】=1时不开方，一个是Max【i】<k时不取模，理解后代码感觉比模板还简单一点
最后第三题贴海报运用了特殊离散化
即在a[i]与a[i-1]相差大于1时要添项，避免漏了中心情况。
最后附上贴海报代码（洛谷3740）
#include<bits/stdc++.h>
using namespace std;
const int N=1e5;
int arr[N][2];
int poster[N<<4];
bool vis[N<<4];
void down(int i){
	if(poster[i]!=0){
		poster[i<<1]=poster[i];
		poster[i<<1|1]=poster[i];
		poster[i]=0;
	}
}
void post(int jobl,int jobr,int id,int l,int r,int i){
	if(jobl<=l&&jobr>=r){
		poster[i]=id;
	}else{
		down(i);
		int mid=(l+r)>>1;
		if(jobl<=mid){
			post(jobl,jobr,id,l,mid,i<<1);
		}
		if(jobr>mid){
			post(jobl,jobr,id,mid+1,r,i<<1|1);
		}
	}
}
int query(int jobl,int jobr,int l,int r,int i){
	if(l==r){
		if(poster[i]!=0&&!vis[poster[i]]){
			vis[poster[i]]=1;
			return 1;
		}else{
			return 0;
		}
	}else{
		down(i);
		int ans=0;
		int mid=(l+r)>>1;
		if(jobl<=mid){
			ans+=query(jobl,jobr,l,mid,i<<1);
		}
		if(jobr>mid){
			ans+=query(jobl,jobr,mid+1,r,i<<1|1);
		}
		return ans;
	}
}
int main(){
	int n,m;
	cin>>n>>m;
	for(int i=1;i<=m;i++){
		cin>>arr[i][0]>>arr[i][1];
	}
	vector<int> a;
	a.push_back(n);
	for(int i=1;i<=m;i++){
		a.push_back(arr[i][0]);
		a.push_back(arr[i][1]);
	}
	sort(a.begin(),a.end());
	int len=a.size();
	for(int i=1;i<len;i++){
		if(a[i]-a[i-1]>1){
			a.push_back(a[i-1]+1);
		}
	}
	sort(a.begin(),a.end());
	a.erase(unique(a.begin(),a.end()),a.end());
	int num=lower_bound(a.begin(),a.end(),n)-a.begin() + 1;
	for(int i=1;i<=m;i++){
		int l=lower_bound(a.begin(),a.end(),arr[i][0])-a.begin() + 1;
		int r=lower_bound(a.begin(),a.end(),arr[i][1])-a.begin() + 1;
		post(l,r,i,1,num,1);
	}
	int ans=query(1,num,1,num,1);
	cout<<ans<<endl;
}
